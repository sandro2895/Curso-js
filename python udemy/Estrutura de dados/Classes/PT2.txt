

#-Classe Criada:

#-Objeto Criado:
usuario1 = Funcionarios()
usuario2 = Funcionarios()  <--Caso faça outro parametro não esquecer de criar outro objeto na classe.
 
#-Criando Parametro:
usuario1.nome = 'Aline'     <--Pode ser criado os parametros  para o objeto mesmo com a classe vazia.
usuario1.sobrenome = 'Souza'
usuario1.idade = 28


print(usuario2.nome, usuario2.sobrenome, usuario2.idade)  <--Pode ser colocado todo em um print
