class Student:
    def __init__(self):
        self.name = 'Doca'
        self.grades = [90, 90, 93, 78, 90]

    def average(self):
        return sum(self.grades) / len(self.grades)  <--divida sum por len a lista para mostra a mÃ©dia do aluno.


student = Student()
print(Student.average(student))
